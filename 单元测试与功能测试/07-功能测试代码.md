class TestAuthFlow:
    """测试完整的登录→访问→登出流程"""

    def test_full_login_flow(self, client, app, admin_token):
        headers = {'Authorization': f'Bearer {admin_token}'}

        # Step 1: 注册新用户
        resp = client.post('/api/register', json={
            'username': 'cashier1', 'password': 'pass123'
        }, headers=headers)
        assert resp.status_code == 200
        new_token = resp.get_json()['data']['token']

        # Step 2: 用新用户登录
        resp = client.post('/api/login', json={
            'username': 'cashier1', 'password': 'pass123'
        })
        assert resp.status_code == 200
        login_data = resp.get_json()['data']
        assert login_data['user']['username'] == 'cashier1'

        # Step 3: 用新token访问受保护接口
        new_headers = {'Authorization': f'Bearer {login_data["token"]}'}
        resp = client.get('/api/products', headers=new_headers)
        assert resp.status_code == 200

        # Step 4: 登出
        resp = client.post('/api/logout', headers=new_headers)
        assert resp.status_code == 200

    def test_unauthorized_access_rejected(self, client):
        """无token访问受保护接口应全部返回401"""
        endpoints = [
            ('/api/products', 'GET'),
            ('/api/suppliers', 'GET'),
            ('/api/purchase-orders', 'GET'),
            ('/api/members', 'GET'),
            ('/api/employees', 'GET'),
            ('/api/warehouses', 'GET'),
            ('/api/accounts', 'GET'),
            ('/api/dashboard/kpi', 'GET'),
        ]
        for url, method in endpoints:
            resp = client.open(url, method=method)
            assert resp.status_code == 401, f'{method} {url} 应返回401'

    def test_wrong_permission_rejected(self, client, app):
        """无权限用户访问需权限接口应返回403"""

class TestPurchaseFlow:
    def test_full_procurement_cycle(self, client, app, admin_token):
        """采购订单→审批→入库的完整流程"""
        headers = {'Authorization': f'Bearer {admin_token}'}

        # 1. 准备基础数据
        client.post('/api/suppliers', json={'supplier_name': '蒙牛集团'}, headers=headers)
        client.post('/api/categories', json={'category_name': '乳制品'}, headers=headers)
        client.post('/api/warehouses', json={'warehouse_name': '常温仓'}, headers=headers)
        client.post('/api/products', json={
            'product_name': '纯牛奶250ml', 'category_id': 1, 'supplier_id': 1,
            'base_price': 5.5, 'cost_price': 3.8
        }, headers=headers)

        # 2. 创建采购订单(草稿)
        resp = client.post('/api/purchase-orders', json={
            'supplier_id': 1, 'status': 'draft', 'order_date': '2026-01-15',
            'delivery_date': '2026-01-20', 'warehouse_id': 1,
            'items': [{'product_id': 1, 'quantity': 500, 'unit_price': 3.5}]
        }, headers=headers)
        assert resp.status_code == 200
        order_id = resp.get_json()['data']['order_id']

        # 3. 查询订单确认总金额
        resp = client.get('/api/purchase-orders', headers=headers)
        orders = resp.get_json()['data']
        order = next(o for o in orders if o['order_id'] == order_id)
        assert order['total_amount'] == 1750  # 500 * 3.5
        assert order['status'] == 'draft'

        # 4. 提交审批(改为pending)
        resp = client.put(f'/api/purchase-orders/{order_id}', json={
            'status': 'pending'
        }, headers=headers)
        assert resp.status_code == 200

        # 5. 审批通过
        resp = client.put(f'/api/purchase-orders/{order_id}/approve', headers=headers)
        assert resp.status_code == 200

        # 6. 验证审批后状态
        resp = client.get(f'/api/purchase-orders?status=approved', headers=headers)
        approved = resp.get_json()['data']
        assert any(o['order_id'] == order_id for o in approved)

        # 7. 入库
        resp = client.post('/api/inbound-records', json={
            'supplier_id': 1, 'warehouse_id': 1,
            'inbound_date': '2026-01-20T10:00:00',
            'items': [{'product_id': 1, 'quantity': 500, 'batch_no': 'B2026-001',
                       'unit_price': 3.5, 'bin_location': 'A-01-03'}]
        }, headers=headers)
        assert resp.status_code == 200

        # 8. 验证库存数量
        resp = client.get('/api/inventory', headers=headers)
        inventory = resp.get_json()['data']
        assert any(i['product_id'] == 1 and i['stock_quantity'] >= 500 for i in inventory)

    def test_cannot_approve_already_approved_order(self, client, app, admin_token):
        """验证已审批订单不可重复审批"""

        client.post('/api/suppliers', json={'supplier_name': '测试'}, headers=headers)
